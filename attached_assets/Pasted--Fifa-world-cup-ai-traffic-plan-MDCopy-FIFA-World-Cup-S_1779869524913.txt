javascriptconst express = require('express');
const app = express();
app.use(express.json());

// Global metrics
let requestCount = 0;
let avgLatency = 50;
let cpuUsage = 20;
let memoryUsage = 30;
let activeServers = 1;

// Simulated database of valid tickets
const validTickets = new Set();
for (let i = 0; i < 100000; i++) {
  validTickets.add(`TICKET_${i}_2026WC`);
}

// Simulate latency increases under load
function getBaseLatency() {
  const loadFactor = requestCount / 1000; // Increases with request count
  const bottleneck = Math.min(loadFactor * 500, 5000); // Max 5 sec delay
  return 50 + bottleneck;
}

// Health endpoint
app.get('/health', (req, res) => {
  res.status(200).json({
    status: 'healthy',
    uptime: process.uptime(),
    metrics: {
      avgLatency,
      cpuUsage,
      memoryUsage,
      activeServers,
      requestsPerSecond: requestCount / 60
    }
  });
});

// Main ticket validation endpoint
app.post('/api/ticket/validate', async (req, res) => {
  const startTime = Date.now();
  requestCount++;

  const { ticketId, userId } = req.body;

  // Simulate latency
  const latency = getBaseLatency();
  await new Promise(resolve => setTimeout(resolve, latency));

  // Simulate crashes at high load
  if (requestCount > 5000 && Math.random() < 0.1) {
    cpuUsage = 95;
    return res.status(503).json({
      error: 'Service overloaded',
      metrics: { cpuUsage, memoryUsage }
    });
  }

  // Process ticket
  const isValid = validTickets.has(ticketId);
  avgLatency = (avgLatency * 0.9) + (Date.now() - startTime) * 0.1;
  cpuUsage = Math.min(95, 20 + (requestCount / 100));
  memoryUsage = Math.min(90, 30 + (requestCount / 150));

  res.status(isValid ? 200 : 401).json({
    valid: isValid,
    processingTime: Date.now() - startTime,
    metrics: { avgLatency, cpuUsage, memoryUsage, activeServers }
  });
});

// Admin endpoint to simulate scaling
app.post('/api/admin/scale', (req, res) => {
  const { action } = req.body;
  
  if (action === 'add-server') {
    activeServers++;
    cpuUsage = Math.max(20, cpuUsage - 15);
    memoryUsage = Math.max(30, memoryUsage - 10);
  } else if (action === 'remove-server' && activeServers > 1) {
    activeServers--;
    cpuUsage = Math.min(95, cpuUsage + 10);
  }

  res.json({
    success: true,
    activeServers,
    metrics: { cpuUsage, memoryUsage, avgLatency }
  });
});

// Reset endpoint (for testing)
app.post('/api/admin/reset', (req, res) => {
  requestCount = 0;
  avgLatency = 50;
  cpuUsage = 20;
  memoryUsage = 30;
  activeServers = 1;
  res.json({ status: 'reset' });
});

const PORT = process.env.API_PORT || 3000;
app.listen(PORT, () => {
  console.log(`Fake FIFA Ticketing API running on port ${PORT}`);
  console.log(`⚽ Ready to simulate 80,000 fans entering the stadium`);
});

javascriptimport http from 'k6/http';
import { check, sleep } from 'k6';
import { Counter, Trend, Gauge } from 'k6/metrics';

// Define custom metrics
const errorCount = new Counter('errors');
const responseTime = new Trend('response_time');
const activeVUS = new Gauge('active_vus');

export const options = {
  stages: [
    // Warm up
    { duration: '10s', target: 100 },
    
    // Gradual increase (like fans arriving)
    { duration: '30s', target: 500 },
    { duration: '30s', target: 1000 },
    { duration: '30s', target: 2000 },
    { duration: '30s', target: 3000 },
    
    // Peak load (like all fans at once)
    { duration: '60s', target: 5000 },
    
    // Sustained peak
    { duration: '30s', target: 5000 },
    
    // Cool down
    { duration: '20s', target: 0 }
  ],
  thresholds: {
    'response_time': ['p(95)<2000', 'p(99)<5000'],
    'errors': ['count<100']
  },
  ext: {
    loadimpact: {
      projectID: 3473834,
      name: 'FIFA Stadium Ticketing Load Test'
    }
  }
};

const API_URL = __ENV.API_URL || 'http://localhost:3000';

export default function () {
  activeVUS.add(__VU); // Track concurrent users

  const ticketId = `TICKET_${__VU % 100000}_2026WC`;
  const payload = JSON.stringify({
    ticketId: ticketId,
    userId: `user_${__VU}`
  });

  const params = {
    headers: { 'Content-Type': 'application/json' },
    timeout: '30s'
  };

  // Main load test
  const response = http.post(
    `${API_URL}/api/ticket/validate`,
    payload,
    params
  );

  // Record metrics
  responseTime.add(response.timings.duration);

  // Check for success
  const success = check(response, {
    'status is 200 or 503': (r) => r.status === 200 || r.status === 503,
    'response time < 2s': (r) => r.timings.duration < 2000,
    'has metrics': (r) => r.json('metrics') !== null
  });

  if (!success) {
    errorCount.add(1);
  }

  sleep(0.1); // Small delay between requests
}

export function handleSummary(data) {
  return {
    'summary.json': JSON.stringify(data),
    stdout: textSummary(data, { indent: ' ', enableColors: true })
  };
}

6.2 Dynatrace Client Wrapper
File: ai-agent/dynatrace-client.js
javascriptconst https = require('https');

class DynatraceClient {
  constructor(config) {
    this.envId = config.envId;
    this.apiToken = config.apiToken;
    this.clusterUrl = config.clusterUrl;
  }

  async makeRequest(path, method = 'GET', body = null) {
    return new Promise((resolve, reject) => {
      const url = new URL(this.clusterUrl + path);
      const options = {
        hostname: url.hostname,
        path: url.pathname + url.search,
        method,
        headers: {
          'Authorization': `Api-Token ${this.apiToken}`,
          'Content-Type': 'application/json'
        }
      };

      const req = https.request(options, (res) => {
        let data = '';
        res.on('data', chunk => data += chunk);
        res.on('end', () => {
          try {
            resolve(JSON.parse(data));
          } catch {
            resolve(data);
          }
        });
      });

      req.on('error', reject);
      if (body) req.write(JSON.stringify(body));
      req.end();
    });
  }

  async getMetrics(metricNames) {
    const path = `/api/v2/metrics/query?metricSelector=${metricNames.join(',')}`;
    const response = await this.makeRequest(path);
    
    // Parse response into usable format
    return {
      latencyP50: response.result?.[0]?.data?.[0]?.values?.[0] || 0,
      latencyP95: response.result?.[1]?.data?.[0]?.values?.[0] || 0,
      latencyP99: response.result?.[2]?.data?.[0]?.values?.[0] || 0,
      errorRate: response.result?.[3]?.data?.[0]?.values?.[0] || 0,
      cpuUsage: response.result?.[4]?.data?.[0]?.values?.[0] || 0,
      memoryUsage: response.result?.[5]?.data?.[0]?.values?.[0] || 0
    };
  }

  async getProblems() {
    const path = `/api/v2/problems?filter=status("open")&sort=severity`;
    const response = await this.makeRequest(path);
    return response.problems || [];
  }

  async getAlerts() {
    const path = `/api/v2/events?filter=type("CUSTOM_ALERT")&sort=-created`;
    const response = await this.makeRequest(path);
    return response.events || [];
  }

  async getEntities(entityType = 'SERVICE') {
    const path = `/api/v2/entities?type=${entityType}`;
    const response = await this.makeRequest(path);
    return response.entities || [];
  }
}

javascriptconst http = require('http');

class AutoFixActions {
  async execute(actionCode) {
    console.log(`Executing action: ${actionCode}`);

    switch (actionCode) {
      case 'add-server':
        return this.addServer();
      case 'remove-server':
        return this.removeServer();
      case 'clear-cache':
        return this.clearCache();
      case 'optimize-queries':
        return this.optimizeQueries();
      case 'restart-service':
        return this.restartService();
      default:
        return { success: false, message: 'Unknown action' };
    }
  }

  async addServer() {
    return new Promise((resolve) => {
      const data = JSON.stringify({ action: 'add-server' });
      const options = {
        hostname: 'localhost',
        port: 3000,
        path: '/api/admin/scale',
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          'Content-Length': data.length
        }
      };

      const req = http.request(options, (res) => {
        res.on('end', () => {
          resolve({
            success: true,
            message: '✅ Added new server instance'
          });
        });
      });

      req.write(data);
      req.end();
    });
  }

  async removeServer() {
    // Similar to addServer but with remove action
    return { success: true, message: '✅ Removed server instance' };
  }

  async clearCache() {
    return { success: true, message: '✅ Cleared application cache' };
  }

  async optimizeQueries() {
    return { success: true, message: '✅ Optimized database queries' };
  }

  async restartService() {
    return { success: true, message: '✅ Restarted ticketing service' };
  }
}
